import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score
from sklearn.datasets import load_iris

# Load iris dataset
iris = load_iris()

# Create dataframe
data = pd.DataFrame(iris.data, columns=iris.feature_names)
data['target'] = iris.target

# Define variables
y = data['sepal length (cm)']
X_uni = data[['petal length (cm)']]

# Create model
model_uni = LinearRegression()

# Train model
model_uni.fit(X_uni, y)

# Predict values
y_pred_uni = model_uni.predict(X_uni)

# Calculate R2 score
r2_uni = r2_score(y, y_pred_uni)

print("R2 Score:", r2_uni)

# Plot graph
plt.figure()
plt.scatter(X_uni, y, color='blue', label='Actual Data')
plt.plot(X_uni, y_pred_uni, color='red', label='Regression Line')

plt.xlabel('Petal Length (cm)')
plt.ylabel('Sepal Length (cm)')
plt.title('Univariate Regression')

plt.legend()
plt.show()

import numpy as np
import matplotlib.pyplot as plt

# Sigmoid function
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# Sample data
x = np.array([1,2,3,4,5])
y = np.array([0,0,1,1,1])

# Simple logistic model
a, b = -4, 1.5

# Prediction
x_new = 6
prob = sigmoid(a + b * x_new)

print("Predicted Probability:", prob)

# Plot
x_vals = np.linspace(0,7,100)
y_vals = sigmoid(a + b * x_vals)

plt.scatter(x, y)
plt.plot(x_vals, y_vals)
plt.xlabel("X")
plt.ylabel("Probability")
plt.title("Logistic Regression")
plt.grid(True)
plt.show()

import numpy as np

# Step function
def step(x):
    return 1 if x >= 0 else 0

# Input and output
X = np.array([[0,0],[0,1],[1,0],[1,1]])
y = np.array([0,0,0,1])

# Weights and bias
w = np.array([1,1])
b = -1.5

# Prediction
for x in X:
    output = step(np.dot(x, w) + b)
    print(x, "->", output)


Experiment 5 Artificial Neural Network (ANN) using Python for XOR logic prediction

import numpy as np

# Sigmoid function
def sigmoid(x):
    return 1/(1+np.exp(-x))

# Input and output
X = np.array([[0,0],[0,1],[1,0],[1,1]])
y = np.array([[0],[1],[1],[0]])

# Random weights
np.random.seed(1)
W = np.random.rand(2,1)

# Training
for i in range(10000):
    output = sigmoid(np.dot(X, W))
    error = y - output
    W += np.dot(X.T, error * output * (1-output)) * 0.1

# Output
print("Predicted Output:")
print(output)

Experiment 8 AdaBoost Classifier

import numpy as np
from sklearn.ensemble import AdaBoostClassifier
from sklearn.preprocessing import LabelEncoder

# Data
X = np.array([['Yes','Good'],['No','Good'],['Yes','Average'],['No','Average']])
y = np.array(['Yes','Yes','No','No'])

# Encoding
le = LabelEncoder()
X = np.array([[le.fit_transform(col) for col in X.T]]).reshape(4,2)
y = le.fit_transform(y)

# Train model
model = AdaBoostClassifier(n_estimators=5)
model.fit(X,y)

# Prediction
print("Output:", model.predict(X))
